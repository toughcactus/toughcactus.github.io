求C(n, k) = n! / (k! * (n - k)!) #

说明
combo(n, k) = C(n, k)
fact(n) = n!

fact.h 


// fact.h

int fact(int);

fact.cpp 

// fact.cpp
#include "fact.h"

int fact(int n)
{
    int p = 1;
    for (int i = 1; i <= n; ++i) {
      p *= i;
    }
    return p;
}

combination.h 

// combination.h

int combo(int n, int k);

combination.cpp 

// combination.cpp

#include "combination.h"
#include "fact.h"

int combo(int n, int k)
{
    return fact(n) / fact(n-k) / fact(k);
}

main.cpp 

// main.cpp

#include <iostream>
#include "combination.h"
#include "fact.h"

using std::cout;
using std::endl;

int main()
{
    for (int i = 1; i < 10; ++i) {
        cout << i << ":" << fact(i) << endl;
    }
    cout << combo(10, 3) << endl;
    return 0;
}

编写makefile #

g++

  1. 查询单个文件依赖 
    g++ -MM main.cpp
  2. 批量查询依赖 
    g++ -MM *cpp
  3. makefile 
    main: main.o fact.o combination.o
    g++ -o main main.o fact.o combination.o

combination.o: combination.cpp combination.h fact.h
    g++ -c combination.cpp

fact.o: fact.cpp fact.h
    g++ -c fact.cpp

main.o: main.cpp combination.h fact.h
    g++ -c main.cpp

Emacs 编译命令 #

(compile)

命令 

make -p
