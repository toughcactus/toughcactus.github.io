fact(n) = n!

fact.h 


// fact.h

int fact(int);

fact.cpp 

// fact.cpp
#include "fact.h"

int fact(int n)
{
    int p = 1;
    for (int i = 1; i <= n; ++i) {
      p *= i;
    }
    return p;
}

combination.h 

// combination.h

int combo(int n, int k);

combination.cpp 

// combination.cpp

#include "combination.h"
#include "fact.h"

int combo(int n, int k)
{
    return fact(n) / fact(n-k) / fact(k);
}

main.cpp 

// main.cpp

#include <iostream>
#include "combination.h"
#include "fact.h"

using std::cout;
using std::endl;

int main()
{
    for (int i = 1; i < 10; ++i) {
        cout << i << ":" << fact(i) << endl;
    }
    cout << combo(10, 3) << endl;
    return 0;
}
