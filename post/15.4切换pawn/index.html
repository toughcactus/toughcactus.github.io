  • 在PlayerController的派生类中完成

一、包含头文件 

  #include "Engine/World.h"
  #include "Kismet/GameplayStatics.h"
  #include "GeometryPawn.h"
  #include "Components/InputComponent.h"

二、添加静态日志类型 

  DEFINE_LOG_CATEGORY_STATIC(LogGeometryPlayerController, All, All)

三、添加数据成员

  • private

      TArray<AActor*>Pawns;
  int32 CurrentPawnIndex = 0;

四、添加函数成员

  • private
  • 获取场景中派生自AGeometryPawn的Pawn对象

  • 在BeginPlay中调用

      void AGeometryPlayerController::InitPawns() {
      UGameplayStatics::GetAllActorsOfClass(GetWorld(), AGeometryPawn::StaticClass(), Pawns);
  }

五、实现BeginPlay 

  void AGeometryPlayerController::BeginPlay() {
      Super::BeginPlay();
      InitPawns();
  }

六、添加函数成员

  • private
  • Pawn个数为1时，切换无效

  • 挨个轮转Pawn对象

      void AGeometryPlayerController::ChangePawn() {
      if (Pawns.Num() <= 1)
      {
          UE_LOG(LogGeometryPlayerController, Log, TEXT("Pawn Num %d"), Pawns.Num());
          return;
      }
      AGeometryPawn *CurrentPawn = Cast<AGeometryPawn>(Pawns[CurrentPawnIndex]);
      CurrentPawnIndex = (CurrentPawnIndex + 1) % Pawns.Num();
      AGeometryPawn *NextPawn = Cast<AGeometryPawn>(Pawns[CurrentPawnIndex]);

      if (!CurrentPawn || !NextPawn) return;

      UE_LOG(LogGeometryPlayerController, Log, TEXT("%s switch pawn from %s to %s"), *GetName(), *CurrentPawn->GetName(), *NextPawn->GetName());

      Possess(NextPawn);
  }

七、添加函数成员

  • private
  • 为函数描述绑定回调函数

  • 在SetupInputComponent中调用

      void AGeometryPlayerController::BindChangePawn() {
      if (InputComponent)
      {
          InputComponent->BindAction("ChangePawn", IE_Pressed, this, &AGeometryPlayerController::ChangePawn);
      }
  }

八、实现SetupInputComponent 

  void AGeometryPlayerController::SetupInputComponent() {
      Super::SetupInputComponent();
      BindChangePawn();
  }
