插件管理

  • 列表安装

      (require 'cl-lib)

  (defvar my/packages '(spacemacs-theme
                        dracula-theme
                        rebecca-theme
                        inkpot-theme
                        farmhouse-themes
                        zeno-theme
                        ) "Default packages") ;; 列出package

  (defun my/packages-installed-p ()
    (cl-loop for pkg in my/packages
             when (not (package-installed-p pkg)) do (cl-return nil) ;; 如果有pkg未安装，返回nil
             finally (cl-return t)
             )) ;; 对列表中的package进行检查

  (unless (my/packages-installed-p) ;; 如果函数调用返回nil
    (message "%s" "Refreshing package database...")
    (package-refresh-contents) ;; 重新拉取package
    (dolist (pkg my/packages)
      (when (not (package-installed-p pkg)) ;; 如果有pkg未安装
        (package-install pkg) ;; 安装
        )))
