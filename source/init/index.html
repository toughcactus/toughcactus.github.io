// init.el
(require 'package)
(setq package-enable-at-startup nil)
(setq package-archives '(("gnu" . "http://mirrors.ustc.edu.cn/elpa/gnu/")
			             ("gnu-devel" . "https://mirrors.ustc.edu.cn/elpa/gnu-devel/")
                         ("melpa" . "http://mirrors.ustc.edu.cn/elpa/melpa/")
                         ("nongnu" . "http://mirrors.ustc.edu.cn/elpa/nongnu/")
			             ("nongnu-devel" . "https://mirrors.ustc.edu.cn/elpa/nongnu-devel/")
			             ("gnu-devel" . "https://mirrors.ustc.edu.cn/elpa/gnu-devel/")
;;			             ("org" . "https://mirrors.ustc.edu.cn/elpa/org/")
			             ("stable-melpa" . "https://mirrors.ustc.edu.cn/elpa/stable-melpa/")
			             ))
(package-initialize)

(unless (package-installed-p 'use-package) ;; 安装use-package
  (package-refresh-contents)
  (package-install 'use-package))

(defun register-windows-super() ;; Windows | 注册Super
  (progn
    (setq w32-pass-lwindow-to-system nil)
    (setq w32-lwindow-modifier 'super)
    (w32-register-hot-key [s-])
    ))

(defun init-emacs(os)
  (if (equal os "macos")
      (progn
	    (setq file-path "/Volumes/Workspace")
	    (setq os-name "macos"))
    (progn
      (setq file-path "W:")
      (setq os-name "windows")
      (register-windows-super)))
  (setq-default init-file "~/.emacs.d/init.el")
  (setq-default my-emacs-file (concat file-path "/post/MyEmacs.org"))
  (setq-default custom-file (expand-file-name (concat file-path "/EmacsConfig/custom_" os-name ".el") user-emacs-directory))
;;  (setq-default org-agenda-files '(concat file-path "/EmacsFiles/Note/"))
  (org-babel-load-file (expand-file-name my-emacs-file))
  (load-file custom-file))

(init-emacs "windows")
(init-emacs "macos")
