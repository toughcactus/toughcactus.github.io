float ASTBaseCharacter::GetAngleBetweenForwardAndVelocity() const
{
	if (GetVelocity().IsZero())
		return 0.0f;

	const FVector ForwardVector = GetActorForwardVector();
	const FVector VelocityNormalizedVector = GetVelocity().GetSafeNormal();

	const float DotProductResult = FVector::DotProduct(ForwardVector, VelocityNormalizedVector);
	const float Theta = FMath::RadiansToDegrees(FMath::Acos(DotProductResult));

	const FVector CrossProductResult = FVector::CrossProduct(ForwardVector, VelocityNormalizedVector);

	return (CrossProductResult.IsZero() ? Theta : FMath::Sign(CrossProductResult.Z) * Theta);
}

void ASTBaseCharacter::OnGroundLanded(const FHitResult& Hit)
{
	float Velocity = -GetVelocity().Z;
	UE_LOG(LogBaseCharacter, Log, TEXT("Velocity: %0.f"), Velocity);

	if (Velocity < LandedDamageVelocityRange.X) return;
	
	float LandedDamage = FMath::GetMappedRangeValueClamped(LandedDamageVelocityRange, LandedDamageRange, Velocity);

	UE_LOG(LogBaseCharacter, Log, TEXT("Damage: %0.f"), LandedDamage);
	TakeDamage(LandedDamage, FDamageEvent{}, nullptr, nullptr);
}

void ASTBaseCharacter::BindOnLanded()
{
	LandedDelegate.AddDynamic(this, &ASTBaseCharacter::OnGroundLanded);
}

{
    const float MaxSpeed = Super::GetMaxSpeed();
    ASTBaseCharacter *Player = Cast<ASTBaseCharacter>(GetPawnOwner());
    return Player && Player->IsRunning() ? SpeedAcceleration * MaxSpeed : MaxSpeed;

void USTHealthComponent::OnTakeAnyDamage(AActor* DamagedActor, float Damage, const class UDamageType* DamageType, class AController* InstigatedBy, AActor* DamageCauser)
{
	if (Damage <= 0.0f || IsDead()) return;

	StopHealTimer();
	// UE_LOG(LogHealthComponent, Display, TEXT("Damage: %.0f"), Damage);
    // Health -= Damage;
	
	SetHealth(Health - Damage);
	// HandleDamage(DamageType);

	InformClientOnDeath();

	InitHealTimer();
}

void USTHealthComponent::RegisterDamageHandler()
{
	AActor *ThisActor = GetOwner();
	if (ThisActor)
	{
		ThisActor->OnTakeAnyDamage.AddDynamic(this, &USTHealthComponent::OnTakeAnyDamage);
	}
}

void USTHealthComponent::HandleDamage(const UDamageType *DamageType)
{
	if (DamageType)
	{
		if (DamageType->IsA<USTFireDamageType>())
		{
			UE_LOG(LogHealthComponent, Display, TEXT("So Hooooooot !!!"));
		}
		else if (DamageType->IsA<USTIceDamageType>())
		{
			UE_LOG(LogHealthComponent, Display, TEXT("So Cooooooooold !!!"));
		}
	}
}

void USTHealthComponent::InformClientOnDeath()
{
	if (IsDead())
	{
		OnDeath.Broadcast();
	}
}

void USTHealthComponent::SetHealth(float NewHealth)
{
	Health = FMath::Clamp(NewHealth, 0.0f, MaxHealth);
	OnHealthChanged.Broadcast();
}

void USTHealthComponent::HealUpdate()
{
	SetHealth(Health + HealModifier);
	if (FMath::IsNearlyEqual(Health, MaxHealth))
	{
		StopHealTimer();
	}
}

void USTHealthComponent::InitHealTimer()
{
	if (!IsDead() && AutoHeal && GetWorld())
	{
		GetWorld()->GetTimerManager().SetTimer(HealTimer, 
												this,
        	                            		&USTHealthComponent::HealUpdate,
            	                    			HealRate, 
												true, 
												HealDelay);
	}
}

void USTHealthComponent::StopHealTimer()
{
	if (AutoHeal && GetWorld())
	{
		GetWorld()->GetTimerManager().ClearTimer(HealTimer);
	}
}
